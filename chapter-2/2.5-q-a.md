Q: What does the following program print?</br>
```cpp
#include <iostream>

void doIt(int x)
{
    int y{ 4 };
    std::cout << "doIt: x = " << x << " y = " << y << '\n';

    x = 3;
    std::cout << "doIt: x = " << x << " y = " << y << '\n';
}

int main()
{
    int x{ 1 };
    int y{ 2 };

    std::cout << "main: x = " << x << " y = " << y << '\n';

    doIt(x);

    std::cout << "main: x = " << x << " y = " << y << '\n';

    return 0;
}
```
A:</br>
main: x = 1 y = 2</br>
doIt: x = 1 y = 4</br>
doIt: x = 3 y = 4</br>
main: x = 1 y = 2</br>
